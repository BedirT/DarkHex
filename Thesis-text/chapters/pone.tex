\section{Probability One State Calculator (PONE)}

\begin{algorithm}
    \caption{PONE}\label{pone}
    \begin{algorithmic}[1]
    \Function{PONE}{s, h}
        \If {White's turn}
            \If {Game is over} \Return False
            \Else \State $h \gets h+1$.
            \EndIf
        \EndIf
        \If {$s$ is Black win}
            \Return True
        \ElsIf {$s$ is White win}
            \Return False
        \Else
            \If {$h == 0$}
                \ForAll {$x \in X$ where $X$ is the set of empty cells}
                    \State \textbf{Ensure:} $(s + B[x], h)$ is legal
                    \If {\Call{PONE}{$s + B[x],h$}} \Return True \EndIf
                \EndFor
            \ElsIf {$h > 0$}
                \ForAll {$x \in X$ where $X$ is the set of empty cells}
                    \State \textbf{Ensure:} $(s + W[x], h-1)$ is legal
                    \State \textbf{Ensure:} $(s + B[x], h)$ is legal
                    \If {\Call{PONE}{$s + W[x],h-1$} and \Call{PONE}{$s + B[x],h$}}
                        \State \Return True
                    \EndIf
                \EndFor
            \EndIf
        \EndIf
        \State \Return False
    \EndFunction
    \end{algorithmic}
\end{algorithm}

\begin{algorithm}
    \caption{Search_PONE}
    \label{s_pone}
    \begin{algorithmic}[1]
    \Function{Search-Pone}{}
    \State{$n \gets$ number of cells}
    \State{$results \gets$ empty dictionary}
    \ForAll{$e$ where $e < n$}
        % \State{\em{Number of empty cells $e = n - (n_b + n_w + h)$}}
        \ForAll{$h$ where $h < n/2$}
            \State $S \gets$ all possible states given $e$ and $h$
            \ForAll{$s \in S$}
                \If{$s$ is legal}
                    \State results$[(s, h)] \gets$ game result with $s, h$
                    \If{\Call{PONE}{$s, h$}}
                        \State results$[(s, h)] \gets$ Black wins
                    \EndIf
                \EndIf
            \EndFor
        \EndFor
    \EndFor
    \EndFunction
    \end{algorithmic}
\end{algorithm}
